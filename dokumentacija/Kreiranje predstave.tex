\textbackslash section*\{Slučaj upotrebe: Kreiranje predstave\}

\textbackslash subsection*\{Kratak opis\} Reditelj dostavlja potrebne
informacije za kreiranje predstave, finansijski službenik evidentira sve
troškove za predstavu (uključujući i probe, plate glumaca, pomoćnog
osoblja, scenografiju, itd.) i unosi ih u sistem za evidenciju troškova.

\textbackslash subsection*\{Akteri\} \textbackslash begin\{enumerate\}
\textbackslash item \textbackslash textbf\{Reditelj\} -\/- pravi
detaljan plan za realizaciju predstave. \textbackslash item
\textbackslash textbf\{Finansijski službenik\} -\/- proverava da li se
troškovi predstave uklapaju u budžet pozorišta. \textbackslash item
\textbackslash textbf\{Sistem\} -\/- izvršava se automatski.
\textbackslash end\{enumerate\}

\textbackslash subsection*\{Preduslovi\}
\textbackslash begin\{enumerate\} \textbackslash item Reditelj je
odlučio da napravi novu predstavu. \textbackslash item Napravljen je
detaljan plan svih potrebnih stvari za realizovanje predstave (određeni
su glumci, datumi za probe, scene, osoblje, itd.). \textbackslash item
Postoji budžet. \textbackslash end\{enumerate\}

\textbackslash subsection*\{Osnovni tok\}
\textbackslash begin\{enumerate\} \textbackslash item Reditelj je
analizirao sve potrebne detalje i napravio plan za realizaciju
predstave. \textbackslash item Dostavlja ih finansijskom službeniku.
\textbackslash item Finansijski službenik razrađuje plan troškova za
glumce, reditelja, scenografiju i pomoćno osoblje. \textbackslash item
Finansijski službenik izračunava sve troškove koji mogu da se dese.
\textbackslash item Finansijski službenik proverava da li se troškovi
uklapaju u budžet. \textbackslash item Finansijski službenik daje
potvrdu da će budžet biti obezbeđen za sve vrste troškova.
\textbackslash end\{enumerate\}

\textbackslash subsection*\{Postuslovi\}
\textbackslash begin\{enumerate\} \textbackslash item Nakon odobrenja od
strane finansijskog službenika, reditelj može da krene u realizaciju
projekta. \textbackslash end\{enumerate\}

\textbackslash subsection*\{Alternativni tokovi\}

\textbackslash subsubsection*\{A1: Finansijski službenik je izračunao da
troškovi prelaze dozvoljen budžet\} \textbackslash begin\{enumerate\}
\textbackslash item Reditelj je obavešten da nema dovoljno sredstava za
realizaciju projekta. \textbackslash item Reditelj pravi potrebne izmene
kako bi se troškovi smanjili. \textbackslash item Reditelj dostavlja
novi plan finansijskom službeniku. \textbackslash item Finansijski
službenik računa troškove za novi plan. \textbackslash item Finansijski
službenik odobrava novi plan. \textbackslash end\{enumerate\}
