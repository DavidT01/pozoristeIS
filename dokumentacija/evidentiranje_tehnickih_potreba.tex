\subsection{Slučaj upotrebe: Evidentiranje tehničkih potreba}

\subsubsection*{Kratak opis}
Tehničko osoblje evidentira tehničke potrebe za predstavu kako bi se obezbedili svi potrebni resursi.

\subsubsection*{Akteri}
\begin{itemize}
    \item \textbf{Tehničko osoblje} -- zaposleni koji žele na jednostavan način da evidentiraju tehničke potrebe pozorišta.
\end{itemize}

\subsubsection*{Preduslovi}
\begin{itemize}
    \item Tehničko osoblje je ulogovano u sistem.
    \item Predstava je registrovana u sistemu.
\end{itemize}

\subsubsection*{Postuslovi}
\begin{itemize}
    \item Tehničke potrebe predstave su sačuvane.
\end{itemize}

\subsubsection*{Osnovni tok}
\begin{enumerate}
    \item Tehničko osoblje bira predstavu za koju želi da evidentira tehničke potrebe.
    \item Sistem prikazuje postojeće tehničke potrebe i resurse dodeljene predstavi.
    \item Tehničko osoblje dodaje nove tehničke potrebe ili ažurira postojeće.
    \item Sistem prikazuje evidentirane potrebe i traži potvrdu.
    \item Tehničko osoblje potvrđuje evidentiranje potreba.
    \item Sistem čuva izmene i obaveštava tehničko osoblje da su potrebe evidentirane.
    \item Ako je potrebno, sistem automatski pokreće slučaj upotrebe \textbf{"Naručivanje opreme"} za nabavku nedostajuće opreme.
\end{enumerate}

\subsubsection*{Podtokovi: /}

\subsubsection*{Alternativni tokovi}

\textbf{A1: Otkazivanje evidentiranja}
\begin{itemize}
    \item Tehničko osoblje odustaje od evidentiranja u koraku 5. Sistem ne čuva izmene i slučaj se završava.
\end{itemize}

\subsubsection*{Specijalni zahtevi: /}

\subsubsection*{Dodatne informacije}
\begin{itemize}
    \item U koraku 3 unose se naziv opreme ili resursa, količina, dodatni opisi i napomena o prioritetu.
\end{itemize}
