\subsection{Slučaj upotrebe: Naručivanje opreme}

\subsubsection*{Kratak opis}
Finansijski službenik naručuje potrebnu opremu za predstavu vodeći računa o raspoloživom budžetu.

\subsubsection*{Akteri}
\begin{itemize}
    \item \textbf{Finansijski službenik} -- zadužen za proveru budžeta i odobravanje nabavki.
\end{itemize}

\subsubsection*{Preduslovi}
\begin{itemize}
    \item Finansijski službenik je ulogovan u sistem.
    \item Postoje evidentirane tehničke potrebe predstave.
    \item Budžet predstave je definisan u sistemu.
\end{itemize}

\subsubsection*{Postuslovi}
\begin{itemize}
    \item Deo ili sve tehničke potrebe predstave su izmirene.
\end{itemize}

\subsubsection*{Osnovni tok}
\begin{enumerate}
    \item Finansijski službenik bira predstavu za koju naručuje opremu.
    \item Sistem prikazuje evidentirane tehničke potrebe.
    \item Finansijski službenik bira potrebe koje želi da izmiri.
    \item Sistem prikazuje narudžbenicu sa ukupnim troškom nabavke.
    \item Finansijski službenik potvrđuje narudžbenicu, ako je trošak u granicama budžeta.
    \item Sistem automatski evidentira transakciju.
    \item Sistem prikazuje potvrdu o uspešnoj narudžbini.
    \item Sistem uklanja tehničke potrebe koje su izmirene.
\end{enumerate}

\subsubsection*{Podtokovi: /}

\subsubsection*{Alternativni tokovi}

\textbf{A1: Trošak prelazi budžet}
\begin{itemize}
    \item Ako u koraku 5 troškovi nisu u granicama budžeta, sistem obaveštava finansijskog službenika da trošak premašuje budžet. Slučaj se vraća na korak 2.
\end{itemize}

\textbf{A2: Plaćanje nije uspešno}
\begin{itemize}
    \item Ako u koraku 6 transakcija ne uspe, sistem obaveštava finansijskog službenika o neuspešnom plaćanju. Slučaj se vraća na korak 2.
\end{itemize}

\subsubsection*{Specijalni zahtevi: /}

\subsubsection*{Dodatne informacije}
\begin{itemize}
    \item U koraku 3 prikazuju se naziv opreme ili resursa, potrebna količina, dodatni opisi, napomena o prioritetu i jedinična cena.
\end{itemize}
