\subsection{Slučaj upotrebe: Unos novog zaposlenog}

\subsubsection*{Kratak opis}
Administrator unosi u sistem novog zaposlenog.

\subsubsection*{Akteri}
\begin{itemize}
    \item \textbf{Adminstrator} -- osoba koja unosi novog zaposlenog.
\end{itemize}
\subsubsection*{Preduslovi}
\begin{itemize}
    \item  Administrator poseduje dokumenta koja je novi zaposleni priložio na konkursu.
\end{itemize}

\subsubsection*{Postuslovi}
\begin{itemize}
    \item U sistem je ubačen novi zaposleni.
\end{itemize}

\subsubsection*{Osnovni tok}
\begin{enumerate}
    \item Administrator pristupa sistemu pozorišta.
    \item Administrator bira opciju unos novog zaposlenog.
    \item Administrator unosi osnovne podatke o zaposlenom (ime, prezime, datum rodjenja, JMBG, prebivalište, br. telefona).
    \item Administrator unosi dodatne informacije (dokumente koji su priloženi na konkursu, godine radnog iskustva, pozicija na kojoj se zaposlio, visina plate, stručna sprema)
    \item Administrator unosi opcione informacije (status invaliditeta, broj dece ...)
    \item Sistem prikazuje sve prethodno unete informacije i traži potvrdu od administratora.
    \item Nakon potvrde sistem prikazuje generisanu službenu e-mail adresu za novog zaposlenog.
    \item Sistem prikazuje poruku o uspešnom unosu novog zaposlenog.
\end{enumerate}

\subsubsection*{Alternativni tokovi}
\textbf{A1: Bez unosa opcionih informacija}
\begin{itemize}
    \item Korak 4. može da se preskoči predje odmah na korak 5. pritiskom dugmeta dalje.
\end{itemize}
