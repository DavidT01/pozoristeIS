\subsection{Slučaj upotrebe: Zakazivanje termina}

\subsubsection*{Kratak opis}
Administrator određeni termin (datum + satnica) dodeljuje određenoj predstavi, označava da li se termin odnosi na probu ili izvedbu predstave.

\subsubsection*{Akteri}
\begin{itemize}
    \item \textbf{Administrator} -- osoba koja vrši zakazivanje termina.
\end{itemize}
\subsubsection*{Preduslovi}
\begin{itemize}
    \item  U sistemu postoji bar 1 predstava.
\end{itemize}

\subsubsection*{Postuslovi}
\begin{itemize}
    \item Ažurirano je stanje izabranog termina.
\end{itemize}

\subsubsection*{Osnovni tok}
\begin{enumerate}
    \item Administrator pristupa sistemu pozorišta.
    \item Administrator bira predstavu za koju hoće da zakaže termin.
    \item Sistem prikazuje dostupne datume.
    \item Administrator bira datum.
    \item Sistem prikazuje dostupnu satnicu za izabrani datum.
    \item Administrator bira satnicu za izabrani datum.
    \item Administrator bira da li se termin odnosi na probu ili izvedbu predstave.
    \item Sistem ažurira stanje izabranog termina.
    \item Sistem prikazuje poruku o uspešnom zakazivanju termina.
\end{enumerate}

\subsubsection*{Podtokovi: /}

\subsubsection*{Alternativni tokovi}
\textbf{A1: Nema slobodnih termina}
\begin{itemize}
    \item Sistem prikazuje poruku da nema slobodnih termina. Prikazuje se dugme OK slučaj se završava.
\end{itemize}
